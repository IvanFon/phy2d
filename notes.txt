#include <iostream>
#include <vector>
#include <functional>

class Vector {
  public:
    int x, y;

    Vector() {
      x = 0;
      y = 0;
    }
};

int main() {
  Vector a;
  Vector b;
  Vector c;
  
  std::vector<std::reference_wrapper<Vector>> things;
  
  things.push_back(a);
  things.push_back(b);
  things.push_back(c);
  
  a.x = 4;
  
  for (int i = 0; i < things.size(); i++) {
    std::cout << i << ": " << things[i].get().x << std::endl;
  }
  
  return 0;
}
